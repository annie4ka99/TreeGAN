 private BufferedImage extractGraphicTwoCoordinates(BufferedImage entireGraphic, String graphicName) { try { Point graphicCoordinates = getPoint(graphicName); if(graphicCoordinates != null) { return entireGraphic.getSubimage(graphicCoordinates.x * graphicWidth, graphicCoordinates.y * graphicHeight, graphicWidth, graphicHeight); } } catch (Exception e) { if(Settings.isDebugModeActivated) { e.printStackTrace(); } } return null; }
