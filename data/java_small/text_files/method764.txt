 private String trimValue(String propertyValue) { if (propertyValue == null || propertyValue.length() == 0) { return ""; } int lastpos = propertyValue.indexOf('#'); if (lastpos == -1) { lastpos = propertyValue.length() - 1; } for(; lastpos >= 0 ; --lastpos ) { char c = propertyValue.charAt(lastpos); if (c != ' ' && c != '\t') { break; } } return propertyValue.substring(0, lastpos + 1); }
